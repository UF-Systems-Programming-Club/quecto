This isn't a super strict grammar with like BNF or anything, just a nice reference for the future when changing the parser.

* means 0 or more times
+ means 1 or more times
| means or
{ } means optional if surrounding

program = (proc)* EOF


proc = proc identifer( param_list ) => ( param_list ) block

statement = (expr       |
             decl       |
             assignment |
             if         |
             while      |
             block      |
             return)   ';'


block = '{' (statement)* '}'
if = if expr block (elif expr block)* (else block)
while = while expr block
decl = identifer : {type} = expr
assignment = identifier = expr
return = return expr
expr = ( '(' expr ')' | number | identifier ) (operator expr)*

identifier = (_|a-z|A-Z) (a-z|A-Z|0-9|_)*
